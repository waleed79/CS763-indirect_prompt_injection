    3. **Numerical-fidelity audit** — the 10-row addendum table values must match the JSON sources to 3 decimal places:
       ```bash
       python << 'PYEOF'
       import json, re

       # Source data
       p5 = json.load(open('logs/ablation_table.json'))
       p7 = json.load(open('logs/ablation_table_gptoss_v7.json'))

       # Llama row keys in Phase 5 ablation_table.json (verify against actual file; common keys per Phase 5 D-02)
       LLAMA_KEYS = {
           "DEF-02":      ["def02_only", "def02"],         # alias-tolerant
           "BERT alone":  ["bert_only", "bert"],
           "Perplexity":  ["perplexity_only", "perplexity"],
           "Imperative":  ["imperative_only", "imperative"],
           "Fingerprint": ["fingerprint_only", "fingerprint"],
           "Fused":       ["fused", "fused_fixed_0.5"],
       }
       GPTOSS_KEYS = {
           ("gpt-oss:20b-cloud", "Fused"):  "gptoss20b_cloud__fused",
           ("gpt-oss:20b-cloud", "DEF-02"): "gptoss20b_cloud__def02",
           ("gpt-oss:120b-cloud","Fused"):  "gptoss120b_cloud__fused",
           ("gpt-oss:120b-cloud","DEF-02"): "gptoss120b_cloud__def02",
       }

       def find_row(d, candidates):
           for c in candidates:
               if c in d: return d[c]
           raise KeyError(f"No key from {candidates} found in {list(d.keys())}")

       # Read addendum table from docs/phase5_honest_fpr.md
       text = open('docs/phase5_honest_fpr.md', encoding='utf-8').read()
       m = re.search(r'## Phase 7 addendum.*?(?=^## |\Z)', text, re.S | re.M)
       assert m, "Addendum section not found"
       table_block = m.group()

       # Parse data rows: lines starting with | Model | Defense |
       def parse_table_rows(block):
           rows = []
           for line in block.splitlines():
               if not line.startswith('|') or '---' in line:
                   continue
               cells = [c.strip() for c in line.strip('|').split('|')]
               if len(cells) < 5: continue
               # Skip the header row (first one with 'Model')
               if cells[0].strip() == 'Model': continue
               rows.append(cells)
           return rows

       rows = parse_table_rows(table_block)
       assert len(rows) == 10, f"expected 10 data rows, got {len(rows)}"

       # Lock the cell format to the canonical Phase 5 §4 style: bare 3-decimal floats
       # like "0.123" — NO percent signs, NO trailing units. If a future Phase 5 prose
       # edit changes the format, this assertion must be updated, NOT silently widened.
       PCT_PATTERN = re.compile(r'%')
       def _parse_cell(cell_val: str, where: str) -> float | None:
           if PCT_PATTERN.search(cell_val):
               errors.append(f"{where}: unexpected '%' in cell {cell_val!r} — Phase 5 §4 format is bare floats")
               return None
           try:
               return float(cell_val.strip())
           except ValueError:
               errors.append(f"{where}: unparseable cell {cell_val!r}")
               return None

       errors = []
       # First 6 rows = llama
       llama_label_order = ["DEF-02", "BERT alone", "Perplexity", "Imperative", "Fingerprint", "Fused"]
       for i, label in enumerate(llama_label_order):
           cells = rows[i]
           model_cell, defense_cell, m1_cell, m2_cell, m3_cell = cells[0], cells[1], cells[2], cells[3], cells[4]
           assert "llama" in model_cell.lower(), f"row {i}: expected llama, got {model_cell!r}"
           expected = find_row(p5, LLAMA_KEYS[label])
           for metric, cell_val in [("per_chunk_fpr", m1_cell), ("answer_preserved_fpr", m2_cell), ("judge_fpr", m3_cell)]:
               table_v = _parse_cell(cell_val, f"llama|{label}|{metric}")
               if table_v is None: continue
               src_v = round(expected[metric], 3)
               if abs(table_v - src_v) > 0.0006:
                   errors.append(f"llama|{label}|{metric}: table={table_v} json={src_v}")

       # Last 4 rows = gpt-oss
       gptoss_order = [("gpt-oss:20b-cloud","Fused"), ("gpt-oss:20b-cloud","DEF-02"),
                       ("gpt-oss:120b-cloud","Fused"), ("gpt-oss:120b-cloud","DEF-02")]
       for j, (model, defense) in enumerate(gptoss_order):
           cells = rows[6+j]
           expected = p7[GPTOSS_KEYS[(model, defense)]]
           for metric, cell_val in [("per_chunk_fpr", cells[2]), ("answer_preserved_fpr", cells[3]), ("judge_fpr", cells[4])]:
               table_v = _parse_cell(cell_val, f"{model}|{defense}|{metric}")
               if table_v is None: continue
               src_v = round(expected[metric], 3)
               if abs(table_v - src_v) > 0.0006:
                   errors.append(f"{model}|{defense}|{metric}: table={table_v} json={src_v}")

       if errors:
           print("FIDELITY ERRORS:")
           for e in errors: print(f"  {e}")
           raise SystemExit(1)
       print(f"OK: all {len(rows)} rows × 3 metrics = 30 cells match JSON sources to 3 decimals")
       PYEOF
       ```
       (Format is LOCKED to bare 3-decimal floats matching Phase 5 §4 — no `%`, no trailing units. If the addendum-table format ever changes, this script must be updated explicitly rather than silently widened to accept both.)
