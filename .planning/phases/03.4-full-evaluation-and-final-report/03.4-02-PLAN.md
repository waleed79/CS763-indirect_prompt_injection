```python
def load_ablation(path: Path) -> pd.DataFrame:
    """Source (c): Phase 3.1 single-LLM ablation. Returns 7-row DataFrame (one per
    defense mode) with columns [defense_mode, model, asr_t1, asr_t2, asr_t3, asr_t4,
    fpr, retrieval_rate, n_queries]. Skips aggregator keys (eval05_aggregation,
    causal_attribution, threshold_sweep, atk08_*, atk09_*) — those go into separate
    tables.
    """
    raw = json.loads(path.read_text())
    SKIP_KEYS = {
        "eval05_aggregation", "causal_attribution", "threshold_sweep",
        "atk08_vs_fused_llama", "atk09_vs_nodefense_llama",
        "atk09_vs_imperative_llama", "atk08_vs_fused_mistral",
    }
    rows = []
    for key, entry in raw.items():
        if key in SKIP_KEYS:
            continue
        if not isinstance(entry, dict):
            continue
        rows.append({
            "defense_mode": entry.get("defense_mode", key),
            "model": entry.get("model", "llama3.2:3b"),
            "asr_t1": entry.get("asr_t1", 0.0),
            "asr_t2": entry.get("asr_t2", 0.0),
            "asr_t3": entry.get("asr_t3", 0.0),
            "asr_t4": entry.get("asr_t4", 0.0),
            "fpr": entry.get("fpr", 0.0),
            "retrieval_rate": entry.get("retrieval_rate", 0.0),
            "n_queries": entry.get("n_queries", 0),
        })
    return pd.DataFrame(rows)


def load_cross_model_matrix(path: Path) -> pd.DataFrame:
    """Source (b): Phase 3.3-07 EVAL-V2-01 matrix. Returns 45-row DataFrame as-is from
    the flat-array JSON (model, defense, tier, asr_overall, asr_tier_native, fpr,
    retrieval_rate)."""
    rows = json.loads(path.read_text())
    df = pd.DataFrame(rows)
    # Normalize tier labels for display (tier1 → T1, tier1b → T1b, etc.)
    tier_display = {"tier1": "T1", "tier1b": "T1b", "tier2": "T2", "tier3": "T3", "tier4": "T4"}
    df["tier"] = df["tier"].map(lambda t: tier_display.get(t, t))
    return df


def load_undefended_baseline(logs_dir: Path) -> pd.DataFrame:
    """Source (a): Phase 2.3 cross-LLM undefended baseline + Phase 2.2 10-query reference.
    Returns N-row DataFrame with columns [model, source, asr_tier1, asr_tier2, paired_asr_tier1,
    paired_asr_tier2, retrieval_rate, n_queries].
    """
    rows = []
    # Phase 2.3 100-query canonical runs (4 models)
    for model_key, label in [
        ("llama", "llama3.2:3b"),
        ("mistral", "mistral:7b"),
        ("gptoss20b", "gpt-oss:20b-cloud"),
        ("gptoss120b", "gpt-oss:120b-cloud"),
    ]:
        path = logs_dir / f"eval_harness_undefended_{model_key}.json"
        if not path.exists():
            continue
        agg = json.loads(path.read_text()).get("aggregate", {})
        rows.append({
            "model": label,
            "source": "Phase 2.3 canonical (n=100)",
            "asr_tier1": agg.get("asr_tier1", 0.0),
            "asr_tier2": agg.get("asr_tier2", 0.0),
            "paired_asr_tier1": agg.get("paired_asr_tier1", 0.0),
            "paired_asr_tier2": agg.get("paired_asr_tier2", 0.0),
            "retrieval_rate": agg.get("retrieval_rate", 0.0),
            "n_queries": agg.get("n_queries", 100),
        })
    # Phase 2.2 10-query frozen reference (llama only) per ROADMAP SC-1
    baseline_path = logs_dir / "attack_baseline.json"
    if baseline_path.exists():
        b = json.loads(baseline_path.read_text())
        # Schema may be flatter — defensive get
        rows.append({
            "model": "llama3.2:3b",
            "source": "Phase 2.2 frozen (n=10)",
            "asr_tier1": b.get("asr_tier1", b.get("aggregate", {}).get("asr_tier1", 0.0)),
            "asr_tier2": b.get("asr_tier2", b.get("aggregate", {}).get("asr_tier2", 0.0)),
            "paired_asr_tier1": b.get("paired_asr_tier1", b.get("aggregate", {}).get("paired_asr_tier1", 0.0)),
            "paired_asr_tier2": b.get("paired_asr_tier2", b.get("aggregate", {}).get("paired_asr_tier2", 0.0)),
            "retrieval_rate": b.get("retrieval_rate", b.get("aggregate", {}).get("retrieval_rate", 0.0)),
            "n_queries": 10,
        })
    return pd.DataFrame(rows)


def build_arms_race_table(
    ablation: pd.DataFrame,
    matrix: pd.DataFrame,
    undefended: pd.DataFrame,
    logs_dir: Path,
) -> pd.DataFrame:
    """Combine all 3 sources into one headline arms race table.

    Per CONTEXT D-09 §2: 'arms race' table = (a) + (b) + (c) sources unified.
    Columns: ['Source', 'Model', 'Defense', 'Tier', 'ASR', 'FPR', 'Retrieval Rate', 'Notes'].

    Adds an 'Adaptive vs Fused' row group from logs/adaptive_fused_llama_s1.json
    (per ablation_table.json key 'atk08_vs_fused_llama' AND fresh adaptive_*.json).

    Per ROADMAP SC-1 note 'On the third defense column': causal column is replaced by
    def02 — the table reflects only what was measured.

    IMPORTANT (W-3 fix): the Defense column MUST go through _display_defense() so the
    raw `bert_only` / `fused_fixed_0.5` / `atk08_vs_fused_llama` keys become writeup-
    friendly labels that contain the substrings the unit tests assert on
    (e.g., `"BERT" in md`).
    """
    rows = []
    # (a) Undefended baseline rows — only llama T1+T2 to keep table compact for §2;
    # full per-LLM detail lives in undefended_baseline.md.
    for _, r in undefended.iterrows():
        if r["model"] != "llama3.2:3b" or r["source"] != "Phase 2.3 canonical (n=100)":
            continue
        for tier_idx in [1, 2]:
            rows.append({
                "Source": "(a) Undefended baseline",
                "Model": r["model"],
                "Defense": _display_defense("no_defense"),
                "Tier": f"T{tier_idx}",
                "ASR": r.get(f"paired_asr_tier{tier_idx}", r.get(f"asr_tier{tier_idx}", 0.0)),
                "FPR": "—",
                "Retrieval Rate": r.get("retrieval_rate", 0.0),
                "Notes": f"Phase 2.3 n={r['n_queries']}",
            })
    # (b) Cross-model matrix rows — all 45 cells
    for _, r in matrix.iterrows():
        rows.append({
            "Source": "(b) Cross-model matrix",
            "Model": r["model"].replace("_", ":") if "_" in r["model"] else r["model"],
            "Defense": _display_defense(r["defense"]),
            "Tier": r["tier"],
            "ASR": r["asr_overall"],
            "FPR": r["fpr"],
            "Retrieval Rate": r["retrieval_rate"],
            "Notes": "Phase 3.3-07 single-seed",
        })
    # (c) Ablation rows — 7 defense modes, 4 tiers each
    for _, r in ablation.iterrows():
        for tier_idx in [1, 2, 3, 4]:
            rows.append({
                "Source": "(c) Single-LLM ablation",
                "Model": r["model"],
                "Defense": _display_defense(r["defense_mode"]),
                "Tier": f"T{tier_idx}",
                "ASR": r.get(f"asr_t{tier_idx}", 0.0),
                "FPR": r["fpr"],
                "Retrieval Rate": r["retrieval_rate"],
                "Notes": f"Phase 3.1 n={r['n_queries']}",
            })
    # Adaptive group (D-03 5th defense category) — read adaptive_fused_llama_s1 for
    # Adaptive vs Fused row, per CONTEXT line 451 arms race row pattern
    adaptive_path = logs_dir / "adaptive_fused_llama_s1.json"
    if adaptive_path.exists():
        ad = json.loads(adaptive_path.read_text()).get("aggregate", {})
        rows.append({
            "Source": "(d) Adaptive attack",
```

```python
def emit_table(df: pd.DataFrame, name: str, output_dir: Path, fmt: str) -> None:
    """Write df to {output_dir}/{name}.md and/or .csv per fmt."""
    if fmt in ("markdown", "both"):
        md = df.to_markdown(index=False, floatfmt=".2f")
        (output_dir / f"{name}.md").write_text(md + "\n", encoding="utf-8")
    if fmt in ("csv", "both"):
        (output_dir / f"{name}.csv").write_text(df.to_csv(index=False), encoding="utf-8")
```
