"""Retriever: ChromaDB vector store + SentenceTransformer encoder (RAG-01).

CRITICAL: ChromaDB defaults to L2 distance. We MUST set cosine explicitly
via configuration={"hnsw": {"space": "cosine"}} (RESEARCH.md Pitfall 1).
Embeddings MUST be normalized (normalize_embeddings=True).
"""
from __future__ import annotations

from typing import Any

import chromadb
from sentence_transformers import SentenceTransformer

from rag.config import Config
from rag.corpus import Passage, chunk_text


class Retriever:
    """Manages embedding, indexing, and retrieval over a ChromaDB collection."""

    def __init__(self, config: Config) -> None:
        self.config = config
        self.encoder = SentenceTransformer(config.embed_model)
        self.client = chromadb.PersistentClient(path=config.chroma_path)
        self.collection = self.client.get_or_create_collection(
            name=config.collection,
            configuration={"hnsw": {"space": "cosine"}},
            metadata={"description": f"NQ {config.corpus_size} passages, {config.chunk_words}-word chunks"},
        )

    def index_passages(self, passages: list[Passage]) -> int:
        """Chunk and embed passages into ChromaDB. Idempotent (Pattern 2).

        Returns the total number of chunks in the collection after indexing.
        Per D-06: chunking is done at index time, stored persistently.
        """
        # Build all chunks with metadata
        all_chunks: list[str] = []
        all_ids: list[str] = []
        all_meta: list[dict[str, Any]] = []

        for passage in passages:
            chunks = chunk_text(
                passage.text,
                chunk_words=self.config.chunk_words,
                overlap_words=self.config.overlap_words,
            )
            for ci, chunk in enumerate(chunks):
                chunk_id = f"chunk_{len(all_chunks):05d}"
                all_chunks.append(chunk)
                all_ids.append(chunk_id)
                all_meta.append({
                    "passage_id": passage.passage_id,
                    "chunk_idx": ci,
                    "source_query": passage.query,
                })

        # Idempotent: skip if already indexed with correct count
        if self.collection.count() == len(all_chunks):
            return self.collection.count()

        # Stale state: drop and rebuild
        if self.collection.count() > 0:
            self.client.delete_collection(self.config.collection)
            self.collection = self.client.create_collection(
                name=self.config.collection,
                configuration={"hnsw": {"space": "cosine"}},
                metadata={"description": f"NQ {self.config.corpus_size} passages"},
            )

        # Embed with normalization (cosine-friendly, Pitfall 1)
        embeddings = self.encoder.encode(
            all_chunks,
            batch_size=32,
            normalize_embeddings=True,
            show_progress_bar=True,
            convert_to_numpy=True,
        )

        # ChromaDB has a batch limit; add in batches of 5000
        batch_size = 5000
        for start in range(0, len(all_chunks), batch_size):
            end = min(start + batch_size, len(all_chunks))
            self.collection.add(
                embeddings=embeddings[start:end].tolist(),
                documents=all_chunks[start:end],
                ids=all_ids[start:end],
                metadatas=all_meta[start:end],
            )

        return self.collection.count()

    def query(self, text: str, top_k: int | None = None) -> list[dict[str, Any]]:
        """Retrieve top-k chunks most similar to query text.

        Returns list of dicts: {chunk_id, score, text, metadata}
        Score is cosine similarity (1 - distance), in [0, 1] range.
        """
        k = top_k or self.config.top_k
        q_emb = self.encoder.encode(
            [text], normalize_embeddings=True, convert_to_numpy=True
        )
        results = self.collection.query(
            query_embeddings=q_emb.tolist(),
            n_results=k,
            include=["documents", "distances", "metadatas"],
        )

        hits: list[dict[str, Any]] = []
        for i in range(len(results["ids"][0])):
            hits.append({
                "rank": i + 1,
                "chunk_id": results["ids"][0][i],
                "score": round(1.0 - results["distances"][0][i], 4),  # cosine sim
                "text": results["documents"][0][i],
                "metadata": results["metadatas"][0][i],
            })
        return hits
```

"""Retrieval logger: append-only JSONL writer (RAG-03).

Per D-07: verbose log entries with query, top-k, doc IDs, scores, full text, timestamp.
Per D-08: logs written to logs/retrieval.jsonl, one JSON object per line, append mode.

Score field: cosine similarity (1 - ChromaDB distance). See RESEARCH.md Pattern 3.
"""
from __future__ import annotations

import json
from datetime import datetime, timezone
from pathlib import Path
from typing import Any


class RetrievalLogger:
    """Append-only JSONL logger for retrieval events."""

    def __init__(self, log_path: str | Path) -> None:
        self.log_path = Path(log_path)
        self.log_path.parent.mkdir(parents=True, exist_ok=True)
        # Line-buffered for crash safety (Pitfall 6)
        self._file = open(self.log_path, "a", encoding="utf-8", buffering=1)

    def log(
        self,
        query: str,
        top_k: int,
        collection: str,
        results: list[dict[str, Any]],
    ) -> None:
        """Write one JSONL record for a retrieval event."""
        record = {
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "query": query,
            "top_k": top_k,
            "collection": collection,
            "results": results,
        }
        self._file.write(json.dumps(record, ensure_ascii=False) + "\n")
        self._file.flush()

    def close(self) -> None:
        """Flush and close the log file."""
        if not self._file.closed:
            self._file.flush()
            self._file.close()

    def __del__(self) -> None:
        try:
            self.close()
        except Exception:
            pass
```

```python
"""LLM Generator: Ollama chat wrapper for RAG context-grounded answers (RAG-01).

Uses ollama.Client (not module-level functions) so tests can inject a mock.
Per RESEARCH.md Pattern 4: system prompt frames context as untrusted data.
Temperature=0.0 + seed for determinism (Pattern 5).
"""
from __future__ import annotations

from ollama import Client, ResponseError

from rag.config import Config


class Generator:
    """Wraps Ollama chat for RAG-grounded generation."""

    SYSTEM_PROMPT = (
        "You are a helpful assistant. Answer the user's question using ONLY "
        "the information in the CONTEXT section below. If the context does "
        "not contain the answer, say so."
    )

    def __init__(self, config: Config) -> None:
        self.config = config
        self.client = Client(host=config.ollama_host)
        self._check_model()

    def _check_model(self) -> None:
        """Verify the target LLM is available in Ollama (Pitfall 4)."""
        try:
            models = self.client.list()
            available = [m.model for m in models.models]
            # Ollama returns names like "llama3.2:3b" or "llama3.2:3b-instruct-..."
            if not any(self.config.llm_model in name for name in available):
                raise RuntimeError(
                    f"Model '{self.config.llm_model}' not found in Ollama. "
                    f"Available: {available}. "
                    f"Run: ollama pull {self.config.llm_model}"
                )
        except Exception as e:
            if "connect" in str(e).lower():
                raise RuntimeError(
                    "Cannot connect to Ollama. Is the server running? "
                    "Run: ollama serve"
                ) from e
            raise

    def generate(self, query: str, context_chunks: list[str]) -> str:
        """Generate an answer grounded in retrieved context chunks.

        Args:
            query: The user's question.
            context_chunks: List of retrieved text chunks (already ranked).

        Returns:
            The LLM's answer string.
        """
        context_block = "\n\n".join(
            f"[{i+1}] {c}" for i, c in enumerate(context_chunks)
        )
        user_prompt = f"CONTEXT:\n{context_block}\n\nQUESTION: {query}"

        try:
            resp = self.client.chat(
                model=self.config.llm_model,
                messages=[
                    {"role": "system", "content": self.SYSTEM_PROMPT},
                    {"role": "user", "content": user_prompt},
                ],
                options={"temperature": 0.0, "seed": self.config.seed},
            )
            return resp["message"]["content"]
        except ResponseError as e:
            if e.status_code == 404:
                raise RuntimeError(
                    f"Model '{self.config.llm_model}' not found. "
                    f"Run: ollama pull {self.config.llm_model}"
                ) from e
            raise
```

**File 2: `rag/pipeline.py`**
```python
"""RAG Pipeline orchestrator: ties Retriever + Generator + Logger (RAG-01).

This is the main entry point. RAGPipeline.query(question) does:
1. Set global seeds (RAG-05)
2. Load and index corpus if needed (RAG-02)
3. Retrieve top-k chunks (RAG-01)
4. Log retrieval event (RAG-03)
5. Generate LLM answer from context (RAG-01)
6. Return {"answer": str, "hits": list[dict]}
"""
from __future__ import annotations

import os
import random
from typing import Any

import numpy as np
import torch

from rag.config import Config, load_config
from rag.corpus import load_nq_passages, chunk_text
from rag.generator import Generator
from rag.logger import RetrievalLogger
from rag.retriever import Retriever


def set_global_seed(seed: int = 42) -> None:
    """Set seeds for all RNG sources for reproducibility (RAG-05, Pattern 5).

    NOTE: HF datasets.shuffle() has its own seed param and does NOT read
    the global numpy seed. Always pass seed= explicitly to shuffle calls.
    """
    os.environ["PYTHONHASHSEED"] = str(seed)
    random.seed(seed)
    np.random.seed(seed)
    torch.manual_seed(seed)
    torch.cuda.manual_seed_all(seed)


class RAGPipeline:
    """End-to-end RAG pipeline: corpus -> embed -> index -> retrieve -> generate.

    Usage:
        config = load_config("config.toml")
        pipe = RAGPipeline(config)
        result = pipe.query("what is the boiling point of water")
        print(result["answer"])
        print(result["hits"])
    """

    def __init__(self, config: Config) -> None:
        self.config = config

        # 1. Set global seeds
        set_global_seed(config.seed)

        # 2. Init components
        self.retriever = Retriever(config)
        self.generator = Generator(config)
        self.logger = RetrievalLogger(config.log_path)

        # 3. Load and index corpus (idempotent)
        passages = load_nq_passages(n=config.corpus_size, seed=config.seed)
        self.retriever.index_passages(passages)

        # Expose collection for test access (test_pipeline_e2e uses pipe.collection.count())
        self.collection = self.retriever.collection

    def query(self, question: str) -> dict[str, Any]:
        """Run a full RAG query: retrieve -> log -> generate -> return.

        Returns:
            {"answer": str, "hits": [{"rank", "chunk_id", "score", "text", "metadata"}, ...]}
        """
        # Retrieve
        hits = self.retriever.query(question, top_k=self.config.top_k)

        # Log (per D-07, D-08)
        self.logger.log(
            query=question,
            top_k=self.config.top_k,
            collection=self.config.collection,
            results=hits,
        )

        # Generate
        context_chunks = [h["text"] for h in hits]
        answer = self.generator.generate(question, context_chunks)

        return {"answer": answer, "hits": hits}

    def close(self) -> None:
        """Clean up resources."""
        self.logger.close()
```
