Write a Python script that generates the notebook:

```python
import nbformat as nbf

nb = nbf.v4.new_notebook()
nb.metadata.kernelspec = {
    "display_name": "Python 3 (rag-security)",
    "language": "python",
    "name": "python3",
}

cells = []

# Cell 1: Title markdown
cells.append(nbf.v4.new_markdown_cell(
    "# RAG Pipeline Demo — CS 763 Final Project\n\n"
    "**Indirect Prompt Injection in Retrieval-Augmented Generation Systems**\n\n"
    "This notebook demonstrates the baseline RAG pipeline:\n"
    "1. Check Ollama is running\n"
    "2. Load configuration\n"
    "3. Load and inspect the NQ corpus\n"
    "4. Initialize the pipeline (embed + index)\n"
    "5. Run a sample query\n"
    "6. Display retrieved chunks with scores\n"
    "7. Display the LLM answer\n"
    "8. Inspect the retrieval log\n\n"
    "> **Environment:** Run inside the `rag-security` conda env (Python 3.11)."
))

# Cell 2: Check Ollama
cells.append(nbf.v4.new_code_cell(
    "# Cell 1: Check Ollama is running with llama3.2:3b\n"
    "from ollama import Client\n\n"
    "client = Client(host='http://localhost:11434')\n"
    "try:\n"
    "    models = client.list()\n"
    "    available = [m.model for m in models.models]\n"
    "    print(f'Ollama is running. Available models: {available}')\n"
    "    assert any('llama3.2:3b' in m for m in available), (\n"
    "        'llama3.2:3b not found. Run: ollama pull llama3.2:3b'\n"
    "    )\n"
    "    print('llama3.2:3b is available.')\n"
    "except Exception as e:\n"
    "    print(f'ERROR: {e}')\n"
    "    print('Make sure Ollama is running: ollama serve')\n"
    "    raise"
))

# Cell 3: Load config
cells.append(nbf.v4.new_code_cell(
    "# Cell 2: Load configuration\n"
    "from rag.config import load_config\n\n"
    "config = load_config('config.toml')\n"
    "print(f'Seed: {config.seed}')\n"
    "print(f'Top-k: {config.top_k}')\n"
    "print(f'Corpus size: {config.corpus_size}')\n"
    "print(f'Chunk words: {config.chunk_words}, Overlap: {config.overlap_words}')\n"
    "print(f'Embed model: {config.embed_model}')\n"
    "print(f'LLM model: {config.llm_model}')\n"
    "print(f'ChromaDB path: {config.chroma_path}')\n"
    "print(f'Collection: {config.collection}')"
))

# Cell 4: Load corpus and show stats
cells.append(nbf.v4.new_code_cell(
    "# Cell 3: Load corpus and show stats\n"
    "from rag.corpus import load_nq_passages, chunk_text\n\n"
    "passages = load_nq_passages(n=config.corpus_size, seed=config.seed)\n"
    "print(f'Loaded {len(passages)} unique passages')\n"
    "print(f'\\nSample passage (ID {passages[0].passage_id}):')\n"
    "print(f'  Query: {passages[0].query}')\n"
    "print(f'  Text:  {passages[0].text[:200]}...')\n\n"
    "# Show chunking stats\n"
    "all_chunks = []\n"
    "for p in passages:\n"
    "    all_chunks.extend(chunk_text(p.text, config.chunk_words, config.overlap_words))\n"
    "print(f'\\nTotal chunks: {len(all_chunks)}')\n"
    "avg_words = sum(len(c.split()) for c in all_chunks) / len(all_chunks)\n"
    "print(f'Avg words/chunk: {avg_words:.1f}')"
))

# Cell 5: Initialize pipeline (embed + index)
cells.append(nbf.v4.new_code_cell(
    "# Cell 4: Initialize pipeline (embed + index — may take ~30s on first run)\n"
    "from rag.pipeline import RAGPipeline\n\n"
    "pipeline = RAGPipeline(config)\n"
    "print(f'Pipeline initialized. Collection has {pipeline.collection.count()} chunks.')"
))

# Cell 6: Run a sample query
cells.append(nbf.v4.new_code_cell(
    "# Cell 5: Run a sample query\n"
    "question = 'what is the boiling point of water'\n"
    "result = pipeline.query(question)\n"
    "print(f'Query: {question}')\n"
    "print(f'Number of hits: {len(result[\"hits\"])}')"
))

# Cell 7: Display retrieved chunks with scores
cells.append(nbf.v4.new_code_cell(
    "# Cell 6: Display retrieved chunks with similarity scores\n"
    "print(f'Retrieved chunks for: \"{question}\"\\n')\n"
    "for hit in result['hits']:\n"
    "    print(f'--- Rank {hit[\"rank\"]} | {hit[\"chunk_id\"]} | Score: {hit[\"score\"]:.4f} ---')\n"
    "    print(hit['text'][:300])\n"
    "    print()"
))

# Cell 8: Display LLM answer
cells.append(nbf.v4.new_code_cell(
    "# Cell 7: Display LLM answer\n"
    "print('=' * 60)\n"
    "print('LLM ANSWER')\n"
    "print('=' * 60)\n"
    "print(result['answer'])"
))

# Cell 9: Show log entry
cells.append(nbf.v4.new_code_cell(
    "# Cell 8: Inspect the retrieval log\n"
    "import json\n"
    "from pathlib import Path\n\n"
    "log_path = Path(config.log_path)\n"
    "if log_path.exists():\n"
    "    lines = log_path.read_text(encoding='utf-8').strip().splitlines()\n"
    "    print(f'Total log entries: {len(lines)}')\n"
    "    last = json.loads(lines[-1])\n"
    "    print(f'\\nLast entry:')\n"
    "    print(f'  Timestamp: {last[\"timestamp\"]}')\n"
    "    print(f'  Query: {last[\"query\"]}')\n"
    "    print(f'  Top-k: {last[\"top_k\"]}')\n"
    "    print(f'  Results: {len(last[\"results\"])} chunks')\n"
    "else:\n"
    "    print(f'Log file not found at {log_path}')"
))

# Cell 10: Cleanup
cells.append(nbf.v4.new_code_cell(
    "# Cell 9: Cleanup\n"
    "pipeline.close()\n"
    "print('Pipeline closed.')"
))

nb.cells = cells

import nbformat
nbformat.write(nb, "demo.ipynb")
print("demo.ipynb created")
```
