"""Unit tests for rag.corpus.chunk_text — RAG-02."""
import pytest

pytestmark = pytest.mark.filterwarnings("ignore::DeprecationWarning")


def _chunk():
    try:
        from rag.corpus import chunk_text
    except ImportError:
        pytest.skip("rag.corpus.chunk_text not yet implemented (Plan 02)")
    return chunk_text


def test_empty_text_returns_empty_list():
    chunk_text = _chunk()
    assert chunk_text("", chunk_words=200, overlap_words=25) == []


def test_short_text_single_chunk():
    chunk_text = _chunk()
    text = "word " * 50  # 50 words, below 200-word budget
    chunks = chunk_text(text.strip(), chunk_words=200, overlap_words=25)
    assert len(chunks) == 1
    assert chunks[0].split() == text.strip().split()


def test_long_text_overlapping_chunks():
    chunk_text = _chunk()
    words = [f"w{i}" for i in range(450)]  # 450 words → expect 3 chunks at 200/25
    text = " ".join(words)
    chunks = chunk_text(text, chunk_words=200, overlap_words=25)
    assert len(chunks) >= 2
    # Each chunk should have at most 200 words
    for c in chunks:
        assert len(c.split()) <= 200
    # Adjacent chunks should share overlap_words words at the boundary
    first_words = chunks[0].split()
    second_words = chunks[1].split()
    assert first_words[-25:] == second_words[:25]


def test_stride_advances_correctly():
    chunk_text = _chunk()
    words = [f"w{i}" for i in range(600)]
    text = " ".join(words)
    chunks = chunk_text(text, chunk_words=200, overlap_words=25)
    # stride = 200 - 25 = 175. ceil(600 / 175) ≈ 4, but the last chunk is emitted once end >= len(words)
    assert 3 <= len(chunks) <= 5
```

"""Non-blocking spot-check: verify word-to-wordpiece ratio for MiniLM.

Phase 2.1 chunker uses a 200-word budget as an approximation of a 256-token
chunk (RESEARCH.md Pitfall 2). This script tokenizes 5 sample English
passages with MiniLM's own tokenizer and prints the observed ratio. If the
ratio exceeds 1.5 on multiple samples, drop `chunk_words` from 200 → 170 in
config.toml.

Run:  conda run -n rag-security python scripts/check_tokenizer_ratio.py
"""
from __future__ import annotations

from transformers import AutoTokenizer

SAMPLES = [
    # 5 short Wikipedia-ish English passages chosen to cover vocabulary variety.
    "Water is a transparent, tasteless, odorless, and nearly colorless chemical substance, which is the main constituent of Earth's hydrosphere and the fluids of all known living organisms.",
    "The 2012 Summer Olympics, officially the Games of the XXX Olympiad and commonly known as London 2012, was an international multi-sport event held in London, United Kingdom, from 27 July to 12 August 2012.",
    "Albert Einstein was a German-born theoretical physicist, widely acknowledged to be one of the greatest and most influential physicists of all time.",
    "The Great Wall of China is a series of fortifications that were built across the historical northern borders of ancient Chinese states and Imperial China as protection against various nomadic groups.",
    "Machine learning is a field of inquiry devoted to understanding and building methods that learn, that is, methods that leverage data to improve performance on some set of tasks.",
]


def main() -> None:
    tok = AutoTokenizer.from_pretrained("sentence-transformers/all-MiniLM-L6-v2")
    ratios = []
    for i, text in enumerate(SAMPLES, 1):
        n_words = len(text.split())
        n_tokens = len(tok.tokenize(text))
        ratio = n_tokens / n_words if n_words else 0.0
        ratios.append(ratio)
        print(f"sample {i}: words={n_words:3d}  tokens={n_tokens:3d}  ratio={ratio:.2f}")
    avg = sum(ratios) / len(ratios)
    print(f"\naverage ratio: {avg:.2f}")
    if avg > 1.5:
        print("WARNING: ratio > 1.5 — consider lowering chunk_words to 170 in config.toml")
    else:
        print("OK: 200-word budget is safely under MiniLM's 512-token max")


if __name__ == "__main__":
    main()
```
