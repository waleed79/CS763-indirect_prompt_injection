def assert_collection_has_all_tiers(collection_name: str) -> None:
    """D-01a: fail fast if any of the 5 tier passage_id ranges is empty."""
    client = chromadb.PersistentClient(path=".chroma")
    c = client.get_collection(collection_name)
    ranges = [
        ("T1",  TIER1_ID_START,  TIER2_ID_START),
        ("T2",  TIER2_ID_START,  TIER3_ID_START),
        ("T3",  TIER3_ID_START,  TIER1B_ID_START),
        ("T1b", TIER1B_ID_START, TIER4_ID_START),
        ("T4",  TIER4_ID_START,  ADAPTIVE_ID_START),
    ]
    for name, lo, hi in ranges:
        res = c.get(
            where={"$and": [{"passage_id": {"$gte": lo}},
                            {"passage_id": {"$lt":  hi}}]},
            limit=1, include=["metadatas"],
        )
        assert len(res["ids"]) > 0, (
            f"D-01a sanity assert failed: {collection_name!r} has no passage in "
            f"{name} range [{lo}, {hi}). Stale collection?"
        )

def add_provenance(output_path: Path) -> None:
    """D-02a + D-03a: enrich JSON post-hoc; run_eval.py hardcodes phase=03.3."""
    data = json.loads(output_path.read_text(encoding="utf-8"))   # encoding mandatory (Pitfall 1)
    data["phase"] = "06"

    error_count = sum(
        1 for r in data.get("results", [])
        if isinstance(r.get("answer"), str) and r["answer"].startswith("[ERROR:")

    data.setdefault("aggregate", {})["error_count"] = error_count
    tmp = output_path.with_suffix(output_path.suffix + ".tmp")
    tmp.write_text(json.dumps(data, indent=2), encoding="utf-8")
    os.replace(tmp, output_path)        # atomic on POSIX + Windows when same dir

**`_summary_v6.json` composition pattern (D-09c)** — derived from `_summary.json` flat-array schema:
```python
# Schema for each cell (matches existing _summary.json verbatim):
# {model, defense, tier, asr_overall, asr_tier_native, fpr, retrieval_rate}

def compose_summary_v6(existing_summary_path: Path,
                       gptoss_runs: list[Path]) -> list[dict]:
    """45 existing cells (verbatim) + 30 new cells extracted from gpt-oss runs."""
    out = list(json.loads(existing_summary_path.read_text(encoding="utf-8")))  # bit-for-bit copy
    for run_path in gptoss_runs:
        data = json.loads(run_path.read_text(encoding="utf-8"))
        agg = data["aggregate"]
        model = data["llm_model"].replace(":", "_")     # canonicalize: "gpt-oss:20b-cloud" -> "gpt-oss_20b-cloud"
        defense = data["defense_mode"]                  # "off", "fused", or "def02"
        # Map "off" -> "no_defense" to match _summary.json convention
        defense = "no_defense" if defense == "off" else defense
        for tier in ["tier1", "tier1b", "tier2", "tier3", "tier4"]:
            asr = agg.get(f"asr_{tier}", 0.0)
            out.append({
                "model": model, "defense": defense, "tier": tier,
                "asr_overall": asr, "asr_tier_native": asr,
                "fpr": agg.get("fpr", 0.0),
                "retrieval_rate": agg.get("retrieval_rate", 0.0),
            })
    return out  # length = 45 + 6*5 = 75 cells; tested in P6-MTX
```

**Divergences from `run_eval_matrix.py`:**
1. **No tier subdivision** — single-pass per cell (D-09a), 6 subprocess calls total, not 30.
2. **Pre-flight sanity** — `assert_collection_has_all_tiers` BEFORE first subprocess (D-01a); no analog in matrix driver.
3. **Post-run JSON mutation** — `add_provenance()` after each subprocess (D-02a, D-03a); matrix driver writes once and walks away.
4. **Two output prefixes** — undefended runs go to `logs/eval_harness_undefended_*_v6.json` (D-02), defended runs to `logs/eval_matrix/<key>__<defense>__all_tiers_v6.json` (D-09b). Matrix driver uses single LOG_DIR.
5. **Composes `_summary_v6.json` post-hoc** (D-09c) — matrix driver does not; `_summary.json` was authored by a separate Phase 03.3-07 script.
